42 /** ***************************************************************************************************
43  * @brief Thread safe STL queue for one consumer thread and one supplier thread. It is used for sending
44  * messages to the thread that does the logging.
45  * @details In the case of multiple consumer threads one might empty the queue after the other checked
46  * it is not empty. This case is accepted as for the moment is not used this way.
47  *****************************************************************************************************/
48 class HOB_LOG_LOCAL message_queue final
49 {
50 public:
51  /** ***********************************************************************************************
52  * @brief The data type of the log that is being stored inside the queue.
53  *************************************************************************************************/
54  using payload = std::pair<std::uint8_t, std::string>;
55 
56  /** ***********************************************************************************************
57  * @brief Checks if the queue has any message stored in it.
58  * @param void
59  * @returns true - queue is empty.
60  * @returns false - queue has at least 1 message.
61  * @throws N/A.
62  *************************************************************************************************/
63  [[nodiscard]] bool is_empty(void) const noexcept;
64 
65  /** ***********************************************************************************************
66  * @brief Emplaces a log at the end of the queue.
67  * @param severity_bit: Bit indicating the type of message that is being logged (see
68  * log1cxx::severity_level).
69  * @param message: The message of the log.
70  * @returns true - the log has been emplaced successfully.
71  * @returns false - the log has been lost.
72  * @throws N/A.
73  *************************************************************************************************/
74  [[nodiscard]] bool emplace(std::uint8_t severity_bit, std::string&& message) noexcept;
75 
76  /** ***********************************************************************************************
77  * @brief Gets the message at the beginning of the list and removes it. If the queue is empty waits
78  * until a message is being emplaced.
79  * @param void

81  * @throws N/A.
82  *************************************************************************************************/
83  [[nodiscard]] std::optional<payload> pop(void) noexcept;
84 
85  /** ***********************************************************************************************

91  void interrupt_wait(void) noexcept;
92 
93 private:
