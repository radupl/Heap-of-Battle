102 template<size_t TEXTURES_COUNT, size_t COMPONENTS_COUNT>
103 TextureInitializer<TEXTURES_COUNT, COMPONENTS_COUNT>::TextureInitializer(
104  const std::array<std::string, TEXTURES_COUNT> filePaths,
105  const std::array<size_t, COMPONENTS_COUNT> textureIndexes,
106  const std::array<SDL_Rect, COMPONENTS_COUNT> destinations,
107  SDL_Renderer* const renderer) noexcept
108  : componentContainer{}
109  , textureContainer{}
110 {
111  size_t index = 0UL;
112 
113  plog_trace("TextureInitializer is being constructed.");
114  for (; index < TEXTURES_COUNT; ++index)
115  {
116  textureContainer[index].load(filePaths[index], renderer);
117  }
118 
119  for (index = 0UL; index < COMPONENTS_COUNT; ++index)
120  {
121  componentContainer[index].updateTexture(textureContainer[textureIndexes[index]]);
122  componentContainer[index].updatePosition(destinations[index]);
123  }
124 }
125 
126 template<size_t TEXTURES_COUNT, size_t COMPONENTS_COUNT>
127 void TextureInitializer<TEXTURES_COUNT, COMPONENTS_COUNT>::draw(SDL_Renderer* const renderer) noexcept
128 {
129  size_t index = 0UL;
130 
131  plog_verbose("Textures are being drawn.");
132  for (; index < COMPONENTS_COUNT; ++index)
133  {
134  componentContainer[index].draw(renderer);
135  }
136 }
