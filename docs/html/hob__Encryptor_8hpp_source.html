43 /** ***************************************************************************************************
44  * @brief Common class allowing for messages to be encrypted and decrypted.
45  * @details This feature is associated to the chat. The host (or the Alliance player) sends its
46  * private key and the other player sends one back. With the help of the Diffie-Hellman key
47  * exchange (https://en.wikipedia.org/wiki/Diffie%E2%80%93Hellman_key_exchange) both players now
48  * have a common key (not known to a middle man) which is used by libobfuscator.so (private
49  * internal HOB library).
50  *****************************************************************************************************/
51 class Encryptor
52 {
53 public:
54  /** ***********************************************************************************************
55  * @brief Produces a random exponent and generates a private key.
56  * @param socket: Sends the private key to the other player.
57  * @return void
58  *************************************************************************************************/
59  void sendKey(const Socket& socket) noexcept;
60 
61  /** ***********************************************************************************************
62  * @brief Uses the private key and the key received from the other player and produces the common
63  * key. In the case this is received without sending one generates one first.
64  * @param enemyKey: The key received from the other player.
65  * @param socket: In the case this is received first send the private key back.
66  * @return void
67  *************************************************************************************************/
68  void receivedKey(uint64_t enemyKey, const Socket& socket) noexcept;
69 
70  /** ***********************************************************************************************
71  * @brief Encrypts a message before sending to the opponent.
72  * @param message: The message to be encrypted.
73  * @return void
74  *************************************************************************************************/
75  void encryptMessage(char* message) noexcept;
76 
77  /** ***********************************************************************************************
78  * @brief Decrypts a message received from the opponent.
79  * @param message: The message to be decrypted.
80  * @return void
81  *************************************************************************************************/
82  void decryptMessage(char* message) noexcept;
83 
84 private:
85  /** ***********************************************************************************************
86  * @brief Determines a private key according to Diffie-Hellman key exchange.
87  * @param base: The base which gets raised to the power of exponent and then divided by modulus.
88  * @return The private key.
89  *************************************************************************************************/
90  uint64_t diffieHellmanKey(const uint64_t base) const noexcept;
91 
92 private:
93  /** ***********************************************************************************************
94  * @brief The common key resulted from the user private key and the other player private key.
95  *************************************************************************************************/
96  uint64_t privateKey;
97 
98  /** ***********************************************************************************************
99  * @brief Random number used for generating the private key.
100  *************************************************************************************************/
101  uint64_t exponent;
102 };
