44 /** ***************************************************************************************************
45  * @brief Thread safe STL queue for one consumer thread and one supplier thread.
46  * @details It is used for updates that are being processed on other threads than the main one. In the
47  * case of multiple consumer threads one might empty the queue after the other checked it is not empty.
48  * This case is accepted as for the moment is not used this way.
49  *****************************************************************************************************/
50 template<typename TYPE>
51 class AsyncQueue final
52 {
53 public:
54  /** ***********************************************************************************************
55  * @brief Checks if the queue has any elements stored in it.
56  * @param void
57  * @return true - queue is empty.
58  * @return false - queue has at least 1 element.
59  *************************************************************************************************/
60  bool isEmpty(void) noexcept;
61 
62  /** ***********************************************************************************************
63  * @brief Inserts an element at the end of the queue.
64  * @param element: The element to be inserted.
65  * @return void
66  *************************************************************************************************/
67  void push(TYPE element) noexcept;
68 
69  /** ***********************************************************************************************
70  * @brief Gets the element at the beginning of the list and removes it. If the queue is empty
71  * (which should not be a valid case) the game will abort() in case of debug builds and crash in
72  * the production builds.
73  * @param void
74  * @return The element.
75  *************************************************************************************************/
76  TYPE pop(void) noexcept;
77 
78 private:
79  /** ***********************************************************************************************
80  * @brief The queue where the elements are being placed.
81  *************************************************************************************************/
82  std::queue<TYPE> queue;
83 
84  /** ***********************************************************************************************
85  * @brief The mutex protecting the queue from multiple threads access.
86  *************************************************************************************************/
87  std::mutex mutex;
88 };

94 template<typename TYPE>
95 bool AsyncQueue<TYPE>::isEmpty(void) noexcept
96 {
97  std::unique_lock<std::mutex> lock(mutex);
98  return queue.empty();
99 }

101 template<typename TYPE>
102 void AsyncQueue<TYPE>::push(TYPE element) noexcept
103 {
104  std::unique_lock<std::mutex> lock(mutex);
105 
106  plog_verbose("Element is being pushed into asynchronically queue.");
107  try
108  {
109  queue.push(element);
110  }
111  catch (const std::bad_alloc& exception)
112  {
113  plog_error("Failed to push element to queue! (bytes: %" PRIu64 ")", sizeof(element));
114  }
115 }
116 
117 template<typename TYPE>
118 TYPE AsyncQueue<TYPE>::pop(void) noexcept
119 {
120  TYPE element;
121 
122  plog_verbose("Element is being popped from asynchronically queue.");
123  plog_assert(false == isEmpty());
124 
125  mutex.lock();
126  element = queue.front();
127  queue.pop();
128  mutex.unlock();
129 
130  return element;
131 }
